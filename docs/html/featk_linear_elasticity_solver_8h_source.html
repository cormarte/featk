87 template<unsigned int Dimension>
88 void featkLinearElasticitySolver<Dimension>::postProcess(const VectorXd& u) {
89 
90  this->mesh->setNodeAttributeFromValues(this->outputAttributeName, 1, u);
91  this->mesh->computeNodeCBQ<1>(this->stiffnessAttributeName, this->outputAttributeName, "Stress Tensor"); // Compute this before moving nodes!
92  this->mesh->addNodeAttributeFromValues("Cartesian Coordinates", 1, u);
93 
94  std::vector<featkNode<3>*> nodes = this->mesh->getNodes();
95  MatrixXd values(nodes.size(), 1);
96 
97  size_t id = this->mesh->getNodeAttributeID("Stress Tensor", 2);
98 
99  for (size_t n=0; n!=nodes.size(); n++) {
100 
101  MatrixXd s = nodes[n]->getAttributeValue(id);
102  values(n, 0) = sqrt(0.5*((s(0,0)-s(1,1))*(s(0,0)-s(1,1)) + (s(1,1)-s(2,2))*(s(1,1)-s(2,2)) + (s(2,2)-s(0,0))*(s(2,2)-s(0,0))) + 3.0*(s(0,1)*s(1,0) + s(1,2)*s(2,1) + s(2,0)*s(0,2)));
103  }
104 
105  this->mesh->setNodeAttributeFromValues("Von Mises Stresses", 0, values);
106 }
