119 template<unsigned int Dimension>
120 VectorXd featkReactionDiffusionSolver<Dimension>::getGlobalSystemVector(const VectorXd &u) {
121 
122  /*
123  The last term of the equation should be integral(Nt(NU)^2) instead of integral(NtN)*U^2 = MU^2.
124  See Mocenni et al. 2011. for handling of polynomial reaction terms in FEM.
125  */
126 
127  size_t id = this->mesh->setNodeAttributeFromValues("tmp", 0, u);
128  VectorXd ru2 = this->getGlobalVectorFromElements(&featkReactionDiffusionSolver<Dimension>::getElementNtCNQNQIntegralVector, {this->mesh->getElementAttributeID(this->reactionAttributeName, 0), id});
129  return this->m*u + this->timeStep*(this->r*u-ru2);
130 
131  //return this->m*u + this->timeStep*this->r*(u-u.cwiseProduct(u));
132 }
