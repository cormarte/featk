130  std::map<size_t, double> allDOFValues = this->essentialBoundaryConditions->getAllDOFValues();
131  std::map<size_t, double> nonZeroDOFValues = this->essentialBoundaryConditions->getNonZeroDOFValues();
132  std::cout << "featkSolverBase: Info: System has " << nonZeroDOFValues.size() << " non-zero essential boundary conditions." << std::endl;
133 
134  for (const auto& pair : allDOFValues) {
135 
136  f(pair.first, 0) = pair.second;
137  }

139  if (nonZeroDOFValues.size()!=0) {
