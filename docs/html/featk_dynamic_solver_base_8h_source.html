163 template<unsigned int Dimension, unsigned int Order>
164 void featkDynamicSolverBase<Dimension, Order>::solve() {
165 
166  cout << "featkDynamicSolverBase: Info: System has " << this->numberOfDOFs << " degrees of freedom." << endl;
167 
168  SparseMatrix<double> globalSystemMatrix = this->getGlobalSystemMatrix();
169  SparseMatrix<double> k = this->getEBCModifiedGlobalSystemMatrix(globalSystemMatrix);
170 
171  cout << "featkDynamicSolverBase: Info: System matrix density is " << globalSystemMatrix.nonZeros() << "/" << this->numberOfDOFs*this->numberOfDOFs << "." << endl;
172 
173  ConjugateGradient<SparseMatrix<double>, Lower|Upper> solver; // Better performances can be achieved with a full matrix, hence 'Lower|Upper'. System matrix is self-adjoint since real symmetric.
174  solver.compute(k);
175 
176  VectorXd u = this->getGlobalInitialVector();
177  VectorXd f = VectorXd(this->numberOfDOFs);
178 
179  for (unsigned int i=0; i!=this->numberOfIterations; i++) {
180 
181  f = this->getGlobalSystemVector(u); // Lower triangular matrices can also be used, then use m.selfadjointView<Lower>() * u and change solver template
182  this->applyEBCToGlobalSystemVector(globalSystemMatrix, f);
183  u = solver.solveWithGuess(f, u).head(this->numberOfDOFs);
184 
185  if (this->doLowerCutoff) {
186 
187  u = (u.array() < this->lowerCutoffValue).select(this->lowerCutoffValue, u);
188  }
189 
190  if (this->doUpperCutoff) {
191 
192  u = (u.array() > this->upperCutoffValue).select(this->upperCutoffValue, u);
193  }
194 
195  cout << "featkDynamicSolverBase: Info: Iteration " << i+1 << "/" << this->numberOfIterations << " solved (" << solver.iterations() << " iterations, error: " << solver.error() << ")." << endl;
196 
197  if (find(this->intermediateProcessIterations.begin(), this->intermediateProcessIterations.end(), i) != this->intermediateProcessIterations.end()) {
198 
199  this->intermediateProcess(u, i);
200  }
201  }
202 
203  cout << "featkDynamicSolverBase: Info: System solved" << endl;
204 
205  this->postProcess(u);
206 
207  cout << "featkDynamicSolverBase: Info: Post processing done" << endl;
208 }
